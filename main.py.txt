from fastapi import FastAPI, UploadFile
from fastapi.responses import StreamingResponse
import pandas as pd
import pdfplumber
import re
import zipfile
from io import BytesIO

app = FastAPI()

# Guardamos aquí la relación orden -> guías
orden_guias = {}

@app.post("/procesar_pdfs")
async def procesar_pdfs(pdfs: list[UploadFile]):
    orden_guias.clear()
    zip_buffer = BytesIO()

    with zipfile.ZipFile(zip_buffer, "w") as zipf:
        for pdf in pdfs:
            texto = ""
            with pdfplumber.open(pdf.file) as p:
                for page in p.pages:
                    if page.extract_text():
                        texto += page.extract_text()

            # guía (10 a 12 dígitos)
            guia_match = re.search(r"\b\d{10,12}\b", texto)
            # orden (9 a 11 dígitos)
            orden_match = re.search(r"\b\d{9,11}\b", texto)

            if not guia_match or not orden_match:
                continue

            guia = guia_match.group()
            orden = orden_match.group()

            orden_guias.setdefault(orden, []).append(guia)

            nuevo_nombre = f"{orden}_{guia}.pdf"
            zipf.writestr(nuevo_nombre, pdf.file.read())

    zip_buffer.seek(0)
    return StreamingResponse(
        zip_buffer,
        media_type="application/zip",
        headers={"Content-Disposition": "attachment; filename=pdfs_renombrados.zip"}
    )


@app.post("/procesar_excel")
async def procesar_excel(excel: UploadFile):
    df = pd.read_excel(excel.file, sheet_name="OFFLINE")

    filas_finales = []

    for _, row in df.iterrows():
        orden = str(row[3]).strip()  # columna D


if orden in orden_guias:
    for guia in orden_guias[orden]:
        nueva = row.copy()
        nueva["Guia"] = guia
        filas_finales.append(nueva)

        else:
            filas_finales.append(row)

    df_final = pd.DataFrame(filas_finales)

    output = BytesIO()
    with pd.ExcelWriter(output, engine="openpyxl") as writer:
        df_final.to_excel(writer, index=False, sheet_name="OFFLINE")

    output.seek(0)
    return StreamingResponse(
        output,
        media_type="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
        headers={"Content-Disposition": "attachment; filename=OFFLINE_COMPLETO.xlsx"}
    )
